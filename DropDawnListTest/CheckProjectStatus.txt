UPDATE CEV
	SET CEV.checkResult = (SELECT TOP 1 P.ASKO),
		CEV.ASKO__STATUS = O.STATUS_ASKO
--SELECT TOP 1 CEV.taskId, CEV.objectUID, CEV.reason, CEV.checkResult, CEV.ASKO__STATUS, P.PROJECTUID, P.ASKO, O.OBJECTUID, O.STATUS_ASKO
FROM dbo.CheckExportValues AS CEV
JOIN dbo.PROJECT_STATUSES AS P ON CEV.objectUID = P.PROJECTUID AND P.ASKO !=0
JOIN dbo.OBJECTS AS O ON CEV.objectUID = O.OBJECTUID AND O.STATUS_ASKO = 'Слетел'
